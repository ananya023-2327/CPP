#include<iostream>
using namespace std;

int main(){
    cout<<"enter numbers of the array 1: \n";
    int r=2, c=3, a[10][10],b[10][10], s[10][10]  ;
    for (int i=0; i<r; i++){
        for(int j=0; j<c; j++){
            cin>>a[i][j];
        }
        cout<<endl;
    }

    cout << "The 1st matrix is:" << endl;
    for (int i = 0; i < r; ++i) {
        for (int j = 0; j < c; ++j) {
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    
    cout<<"\n enter numbers of the array 2: \n";
    for (int k=0; k<r; k++){
        for(int l=0; l<c; l++){
            cin>>b[k][l];
        }
        cout<<endl;
    }
    cout << "The 2nd matrix is:" << endl;
    for (int k= 0; k < r; ++k) {
        for (int l= 0; l < c; ++l) {
            cout << b[k][l]<< " ";
        }
        cout << endl;
    }
    
    for(int i=0; i<r; i++){
        for (int j=0; j<c; j++){
            s[i][j]= a[i][j]+b[i][j];
        }
    }
    
    cout<<"\n The sum of the two matrices is: "<<endl;
    for (int i = 0; i < r; ++i) {
        for (int j = 0; j < c; ++j) {
            cout << s[i][j] << " ";
        }
        cout << endl;
    }
     
    
    
}
